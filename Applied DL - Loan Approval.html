This notebook implements a binaryclassification pipeline to predict loan approval. I start by loading and preprocessing the data (scaling numeric features and encoding categoricals), then split into train/validation/test sets. Next, I define a tunable neural network architecturetwo hidden layers with adjustable size, activation, dropout, optimizers, and learning ratesand compare three hyperparametersearch methods (Random Search, Bayesian Optimization, and Hyperband) using Keras Tuner. After training with early stopping, I evaluate performance on the hold-out test set (loss, accuracy, confusion matrix, classification report). Finally, I apply LIME to generate local, rule-based explanations for individual predictions, interpreting which features most strongly drive approval versus denial.

# Step 2 (continued): model-building function with Dropout
from tensorflow.keras import callbacks

def build_model(hp):
    model = Sequential()
    # First hidden layer + tunable Dropout
    model.add(Dense(
        units=hp.Int('units_0', min_value=32, max_value=128, step=32),
        activation=hp.Choice('activation', ['relu','tanh','sigmoid']),
        input_shape=(X_train_scaled.shape[1],)
    ))
    model.add(Dropout(hp.Float('dropout_rate_0', 0.1, 0.5, step=0.1)))

    # Second hidden layer + same Dropout rate
    model.add(Dense(
        units=hp.Int('units_1', min_value=32, max_value=128, step=32),
        activation=hp.get('activation')
    ))
    model.add(Dropout(hp.get('dropout_rate_0')))

    # Output
    model.add(Dense(1, activation='sigmoid'))

    # Optimizer + learning rate
    opt = hp.Choice('optimizer', ['adam','adagrad','rmsprop'])
    lr  = hp.Choice('learning_rate', [1e-2,1e-3,1e-4])
    if opt=='adam':
        optimizer = Adam(learning_rate=lr)
    elif opt=='adagrad':
        optimizer = Adagrad(learning_rate=lr)
    else:
        optimizer = RMSprop(learning_rate=lr)

    model.compile(optimizer=optimizer,
                  loss='binary_crossentropy',
                  metrics=['accuracy'])
    return model

def evaluate_model(model, X_test, y_test):
    return model.evaluate(X_test, y_test, verbose=0)[0]

from tensorflow.keras.callbacks import EarlyStopping

    patience=3,
    min_delta=0.001,

# Random Search

    max_trials=max_trials,
    directory='random_search',
    project_name='loan_prediction',
    overwrite=True
)
